    from_days

    # from_days

    # description

    # Syntax

    DATE FROM_DAYS(INT N)

    通过距离0000-01-01日的天数计算出哪一天

    # example

    mysql> select from_days(730669);
+-------------------+
| from_days(730669) |
+-------------------+
| 2000-07-03        |
+-------------------+
    1
    2
    3
    4
    5
    6

    # keywords

    FROM_DAYS,FROM,DAYS

    dayofyear from_unixtime
